mdfy API 文档

PDF → Markdown AI 转换服务 · RESTful API v1

Base URL: http://localhost:23504/api/v1
交互式文档（JSON）： /api/v1/docs

认证

如果服务端设置了 MDFY_API_KEY 环境变量，则所有 API 请求（除 /health 和 /docs）需要携带 API Key：

方式示例
HTTP HeaderX-API-Key: your-key-here
Query 参数?api_key=your-key-here

未设置 MDFY_API_KEY 时无需认证，适合本地 / 内网使用。

响应格式

{
  "success": true,          // bool: 请求是否成功
  "data": { ... },          // object|null: 业务数据
  "error": null             // string|null: 错误信息
}

典型工作流

① POST /convert
上传 PDF
→
② GET /tasks/{id}
轮询状态
→
③ GET /tasks/{id}/result
获取结果

也可使用 SSE 端点 /tasks/{id}/progress 替代步骤②，获得实时日志推送。

API 端点

GET /health
健康检查（无需认证）
// Response 200
{
  "success": true,
  "data": { "status": "ok", "timestamp": 1743235200.0 }
}
GET /models
列出可用的 AI 模型
// Response 200
{
  "success": true,
  "data": {
    "models": ["qwen3.5-flash", "qwen3.5-plus"],
    "default": "qwen3.5-plus"
  }
}
POST /convert
上传 PDF 文件并启动异步转换任务

请求参数（multipart/form-data）

参数类型必填说明
fileFile必填PDF 文件
modelString可选AI 模型名，默认 qwen3.5-plus
// Response 202
{
  "success": true,
  "data": {
    "task_id": "20260329_120000_abc123def456",
    "pdf_name": "example.pdf",
    "model": "qwen3.5-plus",
    "status": "pending"
  }
}
GET /tasks/{task_id}
查询任务状态

任务状态值

状态含义
pending已创建，等待处理
converting正在转换中
done转换完成
error转换失败
// Response 200
{
  "success": true,
  "data": {
    "task_id": "20260329_120000_abc123def456",
    "pdf_name": "example.pdf",
    "model": "qwen3.5-plus",
    "status": "converting",
    "elapsed_seconds": 12.3,
    "error": null
  }
}
GET /tasks/{task_id}/progress
SSE 实时进度推送（Server-Sent Events）

事件类型

事件说明
data实时日志行（默认事件类型 message）
done转换完成，payload: {"elapsed": 秒数}
error转换失败，payload: {"error": "错误信息"}
// 使用 JavaScript EventSource
const es = new EventSource("/api/v1/tasks/TASK_ID/progress");
es.onmessage = (e) => console.log("日志:", e.data);
es.addEventListener("done", (e) => {
  console.log("完成:", JSON.parse(e.data));
  es.close();
});
es.addEventListener("error", (e) => {
  console.error("失败:", e.data);
  es.close();
});
GET /tasks/{task_id}/result
获取完整转换结果（Markdown 内容 + 图片列表）
// Response 200
{
  "success": true,
  "data": {
    "task_id": "...",
    "markdown": "# 文档标题\n\n正文内容...\n\n![图片](images/page0_img1.png)",
    "images": ["page0_img1.png", "page1_img1.png"],
    "image_count": 2
  }
}
GET /tasks/{task_id}/result/markdown
仅获取 Markdown 文本（轻量调用）
// Response 200
{ "success": true, "data": { "markdown": "# ..." } }
GET /tasks/{task_id}/images
列出所有生成的图片
// Response 200
{
  "success": true,
  "data": {
    "images": ["page0_img1.png", "page1_img1.png"],
    "count": 2,
    "base_url": "/api/v1/tasks/{task_id}/images/"
  }
}
GET /tasks/{task_id}/images/{filename}
获取指定图片文件（返回图片二进制，Content-Type: image/png）
GET /tasks/{task_id}/download
打包下载 Markdown + 图片（ZIP 格式）
GET /tasks/{task_id}/download/markdown
直接下载 Markdown 文件

状态码

状态码含义
200请求成功
202任务已创建（异步处理中）
400请求参数错误
401认证失败
404资源不存在（任务/图片/文件）
409任务尚未完成，无法获取结果

调用示例

cURL

# 1. 上传 PDF 开始转换
curl -X POST http://localhost:23504/api/v1/convert \
  -H "X-API-Key: your-key" \
  -F "file=@document.pdf" \
  -F "model=qwen3.5-plus"

# 2. 查询任务状态
curl http://localhost:23504/api/v1/tasks/TASK_ID \
  -H "X-API-Key: your-key"

# 3. 获取转换结果
curl http://localhost:23504/api/v1/tasks/TASK_ID/result \
  -H "X-API-Key: your-key"

# 4. 下载 ZIP 包
curl -OJ http://localhost:23504/api/v1/tasks/TASK_ID/download \
  -H "X-API-Key: your-key"

Python

import requests
import time

BASE = "http://localhost:23504/api/v1"
HEADERS = {"X-API-Key": "your-key"}  # 未设置 MDFY_API_KEY 可省略

# 1. 上传 PDF
with open("document.pdf", "rb") as f:
    resp = requests.post(f"{BASE}/convert",
        headers=HEADERS,
        files={"file": f},
        data={"model": "qwen3.5-plus"},
    )
task_id = resp.json()["data"]["task_id"]
print(f"任务已创建: {task_id}")

# 2. 轮询等待完成
while True:
    status = requests.get(f"{BASE}/tasks/{task_id}",
        headers=HEADERS).json()["data"]
    print(f"状态: {status['status']} ({status['elapsed_seconds']}s)")
    if status["status"] == "done":
        break
    if status["status"] == "error":
        raise Exception(status["error"])
    time.sleep(3)

# 3. 获取 Markdown 结果
result = requests.get(f"{BASE}/tasks/{task_id}/result",
    headers=HEADERS).json()["data"]
print(f"转换完成，{result['image_count']} 张图片")
print(result["markdown"][:500])

# 4. 或者直接下载 ZIP
resp = requests.get(f"{BASE}/tasks/{task_id}/download",
    headers=HEADERS)
with open("result.zip", "wb") as f:
    f.write(resp.content)

JavaScript (Node.js)

const FormData = require("form-data");
const fs = require("fs");
const fetch = require("node-fetch");

const BASE = "http://localhost:23504/api/v1";
const HEADERS = { "X-API-Key": "your-key" };

async function convert(pdfPath) {
  // 1. 上传
  const form = new FormData();
  form.append("file", fs.createReadStream(pdfPath));
  form.append("model", "qwen3.5-plus");

  const uploadResp = await fetch(`${BASE}/convert`, {
    method: "POST",
    headers: HEADERS,
    body: form,
  });
  const { data: { task_id } } = await uploadResp.json();
  console.log(`任务已创建: ${task_id}`);

  // 2. 轮询
  while (true) {
    const statusResp = await fetch(`${BASE}/tasks/${task_id}`, { headers: HEADERS });
    const { data } = await statusResp.json();
    if (data.status === "done") break;
    if (data.status === "error") throw new Error(data.error);
    await new Promise(r => setTimeout(r, 3000));
  }

  // 3. 获取结果
  const resultResp = await fetch(`${BASE}/tasks/${task_id}/result`, { headers: HEADERS });
  const { data: result } = await resultResp.json();
  console.log(`完成！${result.image_count} 张图片`);
  return result.markdown;
}

convert("document.pdf").then(md => console.log(md.slice(0, 500)));

SSE 实时进度（JavaScript）

// 浏览器或 Node.js (需 eventsource polyfill)
const es = new EventSource(
  "http://localhost:23504/api/v1/tasks/TASK_ID/progress"
);

es.onmessage = (event) => {
  console.log("[日志]", event.data);
};

es.addEventListener("done", (event) => {
  const { elapsed } = JSON.parse(event.data);
  console.log(`✅ 转换完成，耗时 ${elapsed}s`);
  es.close();
});

es.addEventListener("error", (event) => {
  console.error("❌ 转换失败:", event.data);
  es.close();
});
